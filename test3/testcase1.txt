program example (input, output) ;
var x , y: integer; t, z: char;
function gcd(var a, b: integer; c, d : char): integer;         
begin              
if b=0 then gcd:=a             
else gcd:=gcd(b, a mod b, c, d)         
end;     
begin         
read(x, y);         
write(gcd(x, y, t, z))    
end.
